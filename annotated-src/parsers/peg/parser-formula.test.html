test.each([
  [
    'Ex F(x) -> G(x)',
    'Ex Fx -> Gx'
  ],
  [
    'Ax Ey (F(x, y) -> ~G(y, x))',
    'Ax Ey (Fxy -> ~Gyx)'
  ]
])('\'%s\' is the same as \'%s\'', (text1, text2) => {
  const formula1 = parseFormula(text1)
  const formula2 = parseFormula(text2)
  expect(formula1).toEqual(formula2)
})
